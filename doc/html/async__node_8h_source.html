62 template <>
63 auto Put(base::NodeFragment value, int seq_id, bool final) -> absl::Status {
64 bool explicitly_final = !value.continued && value.seq != -1;
65 bool chunk_is_null =
66 value.chunk.has_value() && base::IsNullChunk(*value.chunk);
67
68 // if the node fragment contains a null chunk, we make sure that it is
69 // marked as final.
70 value.continued = !explicitly_final && !chunk_is_null;
71 return PutFragment(std::move(value), seq_id);
72 }

136// -----------------------------------------------------------------------------
137// IO operators for AsyncNode. These templates have concrete instantiations for
138// Chunk and NodeFragment, and a default overload for all other types, which is
139// implemented in terms of ConstructFrom<Chunk>(T) and MoveAs<T>(Chunk) and
140// therefore specified for types for which these functions are defined.
141// -----------------------------------------------------------------------------
142template <typename T>
143AsyncNode& operator>>(AsyncNode& node, std::optional<T>& value) {
144 std::optional<base::Chunk> chunk;
145 node >> chunk;
146
147 if (!chunk.has_value()) {
148 value = std::nullopt;
149 return node;
150 }
151 value = base::MoveAs<T>(std::move(*chunk));
152 return node;
153}
154
156template <typename T>
157AsyncNode& operator<<(AsyncNode& node, T value) {
158 node.EnsureWriter();
159 return node << std::move(base::Chunk::From(std::move(value)));
160}
161
162// -----------------------------------------------------------------------------
163
164// Concrete instantiation for the operator>> for Chunk.
165template <>
166inline AsyncNode& operator>>(AsyncNode& node,
167 std::optional<base::Chunk>& value) {
168 auto next_chunk_or_status = node.StatusOrNext<base::Chunk>();
169 if (!next_chunk_or_status.ok()) {
170 LOG(ERROR) << "Failed to get next chunk: " << next_chunk_or_status.status();
171 return node;
172 }
173 value = next_chunk_or_status.value();
174 return node;
175}
