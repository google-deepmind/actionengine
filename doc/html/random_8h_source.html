25inline std::string GenerateUUID4() {
26 absl::BitGen gen; // Abseil random number generator
27
28 // Generate 16 random bytes
29 std::array<uint8_t, 16> bytes{};
30 for (int i = 0; i < 16; ++i) {
31 bytes[i] = absl::Uniform<int>(gen, 0, 256); // Generate random byte (0-255)
32 }
33
34 // Set the version (UUID version 4) - The 7th nibble (most significant nibble of 13th byte) must be 4
35 bytes[6] = (bytes[6] & 0x0f) | 0x40;
36
37 // Set the variant (the 9th nibble must be 8, 9, A, or B)
38 bytes[8] = (bytes[8] & 0x3f) | 0x80;
39
40 // Convert to string (UUID format: 8-4-4-4-12)
41 std::ostringstream oss;
42 for (size_t i = 0; i < bytes.size(); ++i) {
43 if (i == 4 || i == 6 || i == 8 || i == 10) {
44 oss << "-";
45 }
46 oss << std::hex << std::setw(2) << std::setfill('0')
47 << static_cast<int>(bytes[i]);
48 }
49
50 return oss.str();
51}
