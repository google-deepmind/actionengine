36template <typename T>
37class ChannelReader {
38 // Implementers of this class should ideally mimic google3's thread::Reader.
39 // The required behaviours are described in method comments below.
40public:
41 // Block until either
42 // (a) the next item has been read into *item; returns true.
43 // (b) Close() has been called on the corresponding writer and
44 // all values have been consumed; returns false.
45 //
46 // When supported, Read() will move into *item.
47 bool Read(T* item [[maybe_unused]]) { return false; }
48
49 // Returns a Case (to pass to Select()) that will finish by either
50 // (a) consuming the next item into *item and storing true in *ok, or
51 // (b) storing false in *ok to indicate that the channel has been closed and
52 // all values have been consumed.
53 //
54 // The objects pointed to by both "item" and "ok" must outlive any call to
55 // Select using this case.
56 //
57 // When supported, OnRead() will move into *item.
58 Case OnRead(T* item [[maybe_unused]], bool* ok [[maybe_unused]]) {
59 return {};
60 }
61};

63template <typename T>
64class ChannelWriter {
65 // Implementers of this class should ideally mimic google3's thread::Writer.
66 // The required behaviours are described in method comments below.
67public:
68 // Blocks until the channel is able to accept a value (for a description of
69 // buffering see Channel()) and writes "item" to the channel.
70 //
71 // See thread::SelectUntil in select.h for how to write only if there's space.
72 //
73 // If "item" is a temporary or the result of calling std::move, it will be
74 // move-assigned into the buffer or a waiting reader.
75 //
76 // REQUIRES: May not be called after this->Close().
77 void Write(const T& item [[maybe_unused]]) {}
78 void Write(T&& item [[maybe_unused]]) {}
79
80 bool WriteUnlessCancelled(const T& item [[maybe_unused]]) { return false; }
81 bool WriteUnlessCancelled(T&& item [[maybe_unused]]) { return false; }
82
83 // Marks the channel as closed, notifying any waiting readers. See
84 // Reader::Read().
85 //
86 // REQUIRES: May be called at most once.
87 void Close();
88};
