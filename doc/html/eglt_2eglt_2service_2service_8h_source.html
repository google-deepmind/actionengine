56class Service : public std::enable_shared_from_this<Service> {
57 // This class is the main entry point for the Evergreen service. It is
58 // responsible for managing sessions, streams, and connections. It also
59 // provides methods to establish and join connections, as well as to set the
60 // action registry.
61 //
62 // The service can be instantiated with an optional action registry and
63 // connection handler. If the action registry is not provided, it will be
64 // initialized with an empty registry. If the connection handler is not
65 // provided, it will be initialized with RunSimpleEvergreenSession. The chunk
66 // store factory is used to create chunk stores for new sessions. By default,
67 // LocalChunkStore-s are created.
68 //
69 // This class is thread-safe. Whenever a connection is joined, it is moved out
70 // of the Service object under a lock, and the Service object is no longer
71 // responsible for managing it. Action registry is set under a lock, and
72 // getters use read locks.
73 //
74 // The Service object can be destroyed at any time, and it will take care of
75 // joining all the connections and cleaning up all the fibers. However, the
76 // Service object itself will not be destroyed until all the connections have
77 // been joined.
78 public:
79 explicit Service(
80 ActionRegistry* action_registry = nullptr,
81 EvergreenConnectionHandler connection_handler = RunSimpleEvergreenSession,
82 ChunkStoreFactory chunk_store_factory = {});
83
84 ~Service();
85
86 Service(const Service& other) = delete;
87 Service& operator=(const Service& other) = delete;

93 auto EstablishConnection(

99 auto SetActionRegistry(const ActionRegistry& action_registry) const -> void;
100
101 private:
102 void JoinConnectionsAndCleanUp(bool cancel = false)
103 ABSL_LOCKS_EXCLUDED(mutex_);
104
105 std::unique_ptr<ActionRegistry> action_registry_;
106 EvergreenConnectionHandler connection_handler_;
107 ChunkStoreFactory chunk_store_factory_;

110 absl::flat_hash_map<std::string, std::shared_ptr<base::EvergreenStream>>
111 streams_ ABSL_GUARDED_BY(mutex_);
112 // for now, we only support one-to-one session-stream mapping, therefore we
113 // use the stream id as the session id.

123 connection_fibers_ ABSL_GUARDED_BY(mutex_);
124
125 bool cleanup_started_ ABSL_GUARDED_BY(mutex_) = false;
126 concurrency::PermanentEvent cleanup_done_;
127};
