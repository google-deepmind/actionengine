44 absl::Status Send(base::SessionMessage message) override {
45 py::gil_scoped_acquire gil;
46 const py::function function = py::get_override(this, "send");
47
48 if (!function) {
49 throw py::error_already_set();
50 }
51 const py::object py_result = function(message);
52
53 const absl::StatusOr<py::object> result =
54 pybindings::RunThreadsafeIfCoroutine(py_result);
55
56 if (!result.ok()) {
57 return result.status();
58 }
59 return absl::OkStatus();
60 }
61
62 std::optional<base::SessionMessage> Receive() override {
63 py::gil_scoped_acquire gil;
64 const py::function function = py::get_override(this, "receive");
65
66 if (!function) {
67 throw py::error_already_set();
68 }
69 const py::object py_result = function();
70
71 absl::StatusOr<py::object> result =
72 pybindings::RunThreadsafeIfCoroutine(py_result);
73
74 if (!result.ok() || result->is_none()) {
75 return std::nullopt;
76 }
77 return std::move(result)->cast<base::SessionMessage>();
78 }
