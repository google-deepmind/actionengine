48class ABSL_SCOPED_LOCKABLE TwoMutexLock {
49 public:
50 explicit TwoMutexLock(Mutex* absl_nonnull mu1, Mutex* absl_nonnull mu2)
51 ABSL_EXCLUSIVE_LOCK_FUNCTION(mu1, mu2)
52 : mu1_(mu1), mu2_(mu2) {
53 if (ABSL_PREDICT_FALSE(mu1_ == mu2_)) {
54 mu1_->Lock();
55 return;
56 }
57
58 if (mu1 < mu2) {
59 mu1_->Lock();
60 mu2_->Lock();
61 } else {
62 mu2_->Lock();
63 mu1_->Lock();
64 }
65 }
66
67 TwoMutexLock(const TwoMutexLock&) = delete; // NOLINT(runtime/mutex)
68 TwoMutexLock(TwoMutexLock&&) = delete; // NOLINT(runtime/mutex)
69 TwoMutexLock& operator=(const TwoMutexLock&) = delete;
70 TwoMutexLock& operator=(TwoMutexLock&&) = delete;
71
72 ~TwoMutexLock() ABSL_UNLOCK_FUNCTION() {
73 if (ABSL_PREDICT_FALSE(mu1_ == mu2_)) {
74 mu1_->Unlock();
75 return;
76 }
77
78 if (mu1_ < mu2_) {
79 mu2_->Unlock();
80 mu1_->Unlock();
81 } else {
82 mu1_->Unlock();
83 mu2_->Unlock();
84 }
85 }
86
87 private:
88 Mutex* absl_nonnull const mu1_;
89 Mutex* absl_nonnull const mu2_;
90};
