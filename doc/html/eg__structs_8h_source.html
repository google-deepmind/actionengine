234struct Chunk {
235 ChunkMetadata metadata;
236
237 std::string ref;
238 std::string data;
239
240 template <typename T>
241 static Chunk From(T&& value) {
242 return ConstructFrom<Chunk>(std::forward<T>(value));
243 }
244
245 template <typename Sink>
246 friend void AbslStringify(Sink& sink, const Chunk& chunk) {
247 if (!chunk.metadata.Empty()) {
248 absl::Format(&sink, "metadata: \n%s",
249 Indent(absl::StrCat(chunk.metadata), 2, true));
250 }
251 if (!chunk.ref.empty() && !chunk.data.empty()) {
252 LOG(FATAL) << "Chunk has both ref and data set.";
253 }
254 if (!chunk.ref.empty()) {
255 absl::Format(&sink, "ref: %s\n", chunk.ref);
256 }
257 if (!chunk.data.empty()) {
258 absl::Format(&sink, "data: %s\n", chunk.data);
259 }
260 }
261};

291 friend void AbslStringify(Sink& sink, const NodeFragment& fragment) {
292 if (!fragment.id.empty()) {
293 absl::Format(&sink, "id: %s\n", fragment.id);
294 }
295 if (fragment.chunk.has_value()) {
296 absl::Format(&sink, "chunk: \n%s",
297 Indent(absl::StrCat(*fragment.chunk), 2, true));
298 }
299 if (fragment.seq != -1) {
300 absl::Format(&sink, "seq: %d\n", fragment.seq);
301 }
302 if (!fragment.continued) {
303 sink.Append("continued: false\n");
304 }
305 if (!fragment.child_ids.empty()) {
306 absl::Format(&sink, "child_ids: %s\n",
307 absl::StrJoin(fragment.child_ids, ", "));
308 }
