43template <typename T, typename S>
44T MoveAs(S value) {
45 absl::StatusOr<T> result = MoveAsAndReturnStatus<T, S>(std::move(value));
46 if (!result.ok()) {
47 LOG(FATAL) << "Failed to move as " << typeid(T).name() << ": "
48 << result.status();
49 }
50 return *std::move(result);

110struct Chunk {
111 ChunkMetadata metadata;
112
113 std::string ref;
114 std::string data;
115
116 template <typename T>
117 static Chunk From(T&& value) {
118 return ConstructFrom<Chunk>(std::forward<T>(value));
119 }
120
121 template <typename Sink>
122 friend void AbslStringify(Sink& sink, const Chunk& chunk) {
123 if (!chunk.metadata.Empty()) {
124 absl::Format(&sink, "metadata: \n%s",
125 Indent(absl::StrCat(chunk.metadata), 2, true));
126 }
127 if (!chunk.ref.empty() && !chunk.data.empty()) {
128 LOG(FATAL) << "Chunk has both ref and data set.";
129 }
130 if (!chunk.ref.empty()) {
131 absl::Format(&sink, "ref: %s\n", chunk.ref);
132 }
133 if (!chunk.data.empty()) {
134 absl::Format(&sink, "data: %s\n", chunk.data);
135 }
136 }
137};

145struct NodeFragment {
147 std::string id;
148
150 std::optional<Chunk> chunk = std::nullopt;
151
153 int32_t seq = -1;
154
156 bool continued = false;
157
159 [[deprecated]] std::vector<std::string> child_ids;
160
161 template <typename T>
162 static NodeFragment From(T&& value) {
163 return ConstructFrom<NodeFragment>(std::forward<T>(value));
164 }
165
166 template <typename Sink>
167 friend void AbslStringify(Sink& sink, const NodeFragment& fragment) {
168 if (!fragment.id.empty()) {
169 absl::Format(&sink, "id: %s\n", fragment.id);
170 }
171 if (fragment.chunk.has_value()) {
172 absl::Format(&sink, "chunk: \n%s",
173 Indent(absl::StrCat(*fragment.chunk), 2, true));
174 }
175 if (fragment.seq != -1) {
176 absl::Format(&sink, "seq: %d\n", fragment.seq);
177 }
178 if (!fragment.continued) {
179 sink.Append("continued: false\n");
180 }
181 if (!fragment.child_ids.empty()) {
182 absl::Format(&sink, "child_ids: %s\n",
183 absl::StrJoin(fragment.child_ids, ", "));
184 }
185 }

222 }
223 if (!action.inputs.empty()) {
224 sink.Append(absl::StrCat("inputs:\n"));
