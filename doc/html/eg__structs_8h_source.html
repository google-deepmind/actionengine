174template <typename T, typename S>
175T MoveAs(S value) {
176 absl::StatusOr<T> result = MoveAsAndReturnStatus<T, S>(std::move(value));
177 if (!result.ok()) {
178 LOG(FATAL) << "Failed to move as " << typeid(T).name() << ": "
179 << result.status();
180 }
181 return *std::move(result);
182}
