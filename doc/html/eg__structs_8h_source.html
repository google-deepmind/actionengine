230struct Chunk {
231 ChunkMetadata metadata;
232
233 std::string ref;
234 std::string data;
235
236 template <typename T>
237 static Chunk From(T&& value) {
238 return ConstructFrom<Chunk>(std::forward<T>(value));
239 }
240
241 template <typename Sink>
242 friend void AbslStringify(Sink& sink, const Chunk& chunk) {
243 if (!chunk.metadata.Empty()) {
244 absl::Format(&sink, "metadata: \n%s",
245 Indent(absl::StrCat(chunk.metadata), 2, true));
246 }
247 if (!chunk.ref.empty() && !chunk.data.empty()) {
248 LOG(FATAL) << "Chunk has both ref and data set.";
249 }
250 if (!chunk.ref.empty()) {
251 absl::Format(&sink, "ref: %s\n", chunk.ref);
252 }
253 if (!chunk.data.empty()) {
254 absl::Format(&sink, "data: %s\n", chunk.data);
255 }
256 }
257};

265struct NodeFragment {
267 std::string id;
268
270 std::optional<Chunk> chunk = std::nullopt;

273 int32_t seq = -1;
274
276 bool continued = false;
277
278 template <typename T>
279 static NodeFragment From(T&& value) {
280 return ConstructFrom<NodeFragment>(std::forward<T>(value));
281 }
282
283 template <typename Sink>
284 friend void AbslStringify(Sink& sink, const NodeFragment& fragment) {
285 if (!fragment.id.empty()) {
286 absl::Format(&sink, "id: %s\n", fragment.id);
287 }
288 if (fragment.chunk.has_value()) {
289 absl::Format(&sink, "chunk: \n%s",
290 Indent(absl::StrCat(*fragment.chunk), 2, true));
291 }
292 if (fragment.seq != -1) {
293 absl::Format(&sink, "seq: %d\n", fragment.seq);
294 }
295 if (!fragment.continued) {
296 sink.Append("continued: false\n");
297 }
298 }
