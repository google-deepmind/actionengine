33class ChunkStoreReader {
34 // This class is thread-compatible. Calls to Run() and Reset() must be
35 // externally synchronized and coordinated with calls to Next() / GetStatus().
36 // In particular, status' reads/updates are protected, but users should not
37 // assume that a call to GetStatus() will return the same status as the
38 // previous call to Run(). Internally, status is updated in a background
39 // thread, so it is possible for the status to change arbitrarily between
40 // various calls.
41 // TODO(helenapankov): clarify the thread-compatibility contract and the
42 // behavior of GetStatus().
43 public:
44 constexpr static float kDefaultWaitTimeout = -1;
45 constexpr static float kNoTimeout = -1;
46
47 explicit ChunkStoreReader(ChunkStore* absl_nonnull chunk_store,
48 bool ordered = false, bool remove_chunks = false,
49 int n_chunks_to_buffer = -1,
50 float timeout = kDefaultWaitTimeout);
51 ~ChunkStoreReader();

53 ChunkStoreReader(const ChunkStoreReader& other) = delete;
54 ChunkStoreReader& operator=(const ChunkStoreReader& other) = delete;

56 absl::Status Run();

59 std::optional<T> Next() {

107template <>
108inline std::optional<std::pair<int, base::Chunk>> ChunkStoreReader::Next() {
109 if (fiber_ == nullptr) {
110 Run().IgnoreError();
111 }
112 std::optional<std::pair<int, base::Chunk>> seq_and_chunk;
113 bool ok;
114 auto deadline = (timeout_ > 0) ? absl::Now() + absl::Seconds(timeout_)
115 : absl::InfiniteFuture();
116 int selected = concurrency::SelectUntil(
117 deadline, {buffer_->GetReader()->OnRead(&seq_and_chunk, &ok)});
118 if (selected == -1) {
119 UpdateStatus(absl::DeadlineExceededError("Timed out waiting for chunk."));
120 return std::nullopt;
121 }
122 UpdateStatus(absl::OkStatus());
123 if (!ok) {
124 return std::nullopt;
125 }
126 if (base::IsNullChunk(seq_and_chunk->second)) {
127 return std::nullopt;
128 }
129 return seq_and_chunk;
130}

150 while (true) {

153 break;
154 }
155 if (!reader.GetStatus().ok()) {
156 LOG(ERROR) << "Error: " << reader.GetStatus() << "\n";
157 break;
158 }
159 value.push_back(std::move(*chunk));
160 }
161 return reader;

184class ChunkStoreWriter {
185 // This class is thread-safe. Public methods can be called concurrently from
186 // different threads. Chunk store and buffer are only accessed by the writer
187 // thread, and are only set on construction. Other fields are only accessed by
188 // the internal writer fiber, so access is always synchronous.
189 public:
190 explicit ChunkStoreWriter(ChunkStore* chunk_store,
191 int n_chunks_to_buffer = -1);
192 ~ChunkStoreWriter() ABSL_LOCKS_EXCLUDED(mutex_);
193
194 ChunkStoreWriter(const ChunkStoreWriter& other) = delete;
195 ChunkStoreWriter& operator=(const ChunkStoreWriter& other) = delete;
196
197 template <typename T>
198 absl::StatusOr<int> Put(T value, int seq = -1, bool final = false)
199 ABSL_LOCKS_EXCLUDED(mutex_) {
200 return Put(base::Chunk::From(std::move(value)), seq, final);
201 }
202
203 // putting a chunk is considered a base case, therefore the definition is
204 // inside the class body.
205 template <>
206 absl::StatusOr<int> Put(base::Chunk value, int seq, bool final)
207 ABSL_LOCKS_EXCLUDED(mutex_) {
208 concurrency::MutexLock lock(&mutex_);
209 if (!accepts_puts_) {
210 return absl::FailedPreconditionError(
211 "Put was called on a writer that is not accepting more puts.");
212 }
213
214 int written_seq = seq;
215 if (seq == -1) {
216 written_seq = total_chunks_put_;
217 }
218 total_chunks_put_++;
219
220 EnsureWriteLoop();
221 max_seq_put_ = std::max(max_seq_put_, written_seq);
222 if (!buffer_->GetWriter()->WriteUnlessCancelled(base::NodeFragment{
223 .chunk = std::move(value),
224 .seq = written_seq,
225 .continued = !final,
226 })) {
227 return absl::CancelledError("Cancelled.");
228 }
229
230 return written_seq;
231 }
232
233 void FinishWrites() ABSL_LOCKS_EXCLUDED(mutex_);
234
235 absl::Status GetStatus() const ABSL_LOCKS_EXCLUDED(mutex_) {
236 concurrency::MutexLock lock(&mutex_);
237 return status_;
238 }

240 template <typename T>
241 friend ChunkStoreWriter& operator<<(ChunkStoreWriter& writer, T value) {
242 writer.Put(base::Chunk::From(std::move(value))).IgnoreError();
243 return writer;
244 }

246 private:
247 void EnsureWriteLoop() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mutex_);
248
249 void SafelyCloseBuffer() ABSL_EXCLUSIVE_LOCKS_REQUIRED(mutex_);

251 void RunWriteLoop();

253 void TerminateSafely() ABSL_LOCKS_EXCLUDED(mutex_);
254

256 concurrency::MutexLock lock(&mutex_);
257 status_ = status;
258 }

260 ChunkStore* chunk_store_ = nullptr;
261 const int n_chunks_to_buffer_;

264 int total_chunks_put_ ABSL_GUARDED_BY(mutex_) = 0;
265
266 bool accepts_puts_ ABSL_GUARDED_BY(mutex_) = true;

276 mutable concurrency::Mutex mutex_;
