180 // TODO(helenapankov): just use hash maps instead of vectors
181 const auto it = std::find_if(
182 def_.inputs.begin(), def_.inputs.end(),
183 [name](const ActionNode& node) { return node.name == name; });
184 if (it == def_.inputs.end()) {
185 return nullptr;
186 }
187
188 AsyncNode* node = GetNode(GetInputId(name));
189 if (stream_ != nullptr &&
190 bind_stream.value_or(bind_streams_on_inputs_default_)) {
191 node->BindWriterStream(stream_);
