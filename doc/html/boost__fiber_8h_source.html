33static std::atomic<int32_t> last_rand32;
34static absl::once_flag init_rand32_once;
35
36static void InitRand32() {
37 // GoogleOnceInit is an acquire barrier on remote-cpus.
38 uint32_t seed = absl::Uniform<uint32_t>(absl::BitGen());
39 // Avoid 0 which generates a sequence of 0s.
40 if (seed == 0)
41 seed = 1;
42 last_rand32.store(static_cast<int32_t>(seed), std::memory_order_release);
43}
44
45// Pseudo-random number generator using Linear Shift Feedback Register (LSFB)
46static uint32_t Rand32() {
47 // Primitive polynomial: x^32+x^22+x^2+x^1+1
48 static constexpr uint32_t poly = (1 << 22) | (1 << 2) | (1 << 1) | (1 << 0);
49
50 absl::call_once(init_rand32_once, InitRand32);
51 uint32_t r = last_rand32.load(std::memory_order_relaxed);
52 r = (r << 1) ^
53 ((static_cast<int32_t>(r) >> 31) & poly); // shift sign-extends
54 last_rand32.store(static_cast<int32_t>(r), std::memory_order_relaxed);
55 return r;
56}

194 MutexLock event_lock(&mutex_);
195 MutexLock selector_lock(&case_state->selector->mutex);

220 // Implementers of this class should ideally mimic google3's thread::Reader.
221 // The required behaviours are described in method comments below.
222 public:
223 // Block until either
224 // (a) the next item has been read into *item; returns true.
225 // (b) Close() has been called on the corresponding writer and
226 // all values have been consumed; returns false.
227 //
228 // When supported, Read() will move into *item.
229 bool Read(T* item [[maybe_unused]]) { return false; }
230
231 // Returns a Case (to pass to Select()) that will finish by either
232 // (a) consuming the next item into *item and storing true in *ok, or
233 // (b) storing false in *ok to indicate that the channel has been closed and
234 // all values have been consumed.
235 //
236 // The objects pointed to by both "item" and "ok" must outlive any call to
237 // Select using this case.
238 //
239 // When supported, OnRead() will move into *item.
240 Case OnRead(T* item [[maybe_unused]], bool* ok [[maybe_unused]]) {
241 return {};
242 }

398inline bool CvBlock(const absl::Time deadline, Selector* sel)
399 ABSL_SHARED_LOCKS_REQUIRED(sel->mutex) {
400 // We must first check that no notification occurred between registration
401 // with Handle and reaching here.
402 while (sel->picked == Selector::kNonePicked) {
403 // Note that implementations of WaitGeneric below are allowed to return
404 // true (indicating timeout) when racing with Signal(). To handle this we
405 // re-check against sel.picked before returning expiring_index.
406 if (sel->cv.WaitWithDeadline(&sel->mutex, deadline) &&
407 sel->picked == Selector::kNonePicked) {
408 return false;
409 }
410 }
411 return true;
412}

414inline int SelectUntil(const absl::Time deadline, const CaseArray& cases) {
415 Selector sel;
416 sel.picked = Selector::kNonePicked;
417 const auto num_cases = cases.size();
418
419 // Initialize internal representation of passed Cases
420 absl::FixedArray<CaseState, 4> case_states(num_cases);
421
422 // Use inside-out Fisher-Yates shuffle to combine initialization and
423 // permutation.
424 if (num_cases > 0) {
425 case_states[0].index = 0;
426 }
427 for (int i = 1; i < num_cases; i++) {
428 const uint32_t swap = Rand32() % (i + 1);
429 case_states[i].index = case_states[swap].index;
430 case_states[swap].index = i;
431 }
432
433 const bool blocking = deadline != absl::InfinitePast();
434 bool ready = false;
435 int registered_limit;
436 for (registered_limit = 0; registered_limit < num_cases; registered_limit++) {
437 CaseState* case_state = &case_states[registered_limit];
438 const Case* assoc_case = &cases[case_state->index];
439 case_state->params = assoc_case;
440 case_state->prev = nullptr; // Not on any list.
441 case_state->selector = &sel;
442 if (assoc_case->selectable->Handle(case_state, /*enqueue=*/blocking)) {
443 ready = true;
444 break;
445 }
446 }
447
448 if (!blocking) {
449 // Do not wait. Also, no need to Unregister() any cases since
450 // we passed enqueue=false to each Handle() above.
451 const int picked = ready ? sel.picked : -1;
452 return picked;
453 }
454
455 if (!ready) {
456 const bool expirable = deadline != absl::InfiniteFuture();
457
458 MutexLock l(&sel.mutex);
459 const bool expired = !CvBlock(deadline, &sel);
460 DCHECK(expirable || !expired);
461 if (expired) {
462 // Deadline expiry. Ensure nothing is picked from this point.
463 sel.picked = num_cases;
464 }
465 }
466
467 // Unregister from all events with which we are registered. We know that
468 // there was no non-blocking index and that we attempted to enqueue against
469 // all cases with index smaller than registered_limit.
470 for (int i = 0; i < registered_limit; i++) {
471 if (CaseState* case_state = &case_states[i];
472 case_state->index != sel.picked) {
473 // sel.picked was unregistered by the notifier.
474 case_state->params->selectable->Unregister(case_state);
475 }
476 }
477
478 // sel.picked == num_cases denotes expiry
479 const int picked = sel.picked < num_cases ? sel.picked : -1;
480 return picked;
481}
